# Preprocessing: Select relevant columns for analysis
tornado_data_clean <- tornado_data %>%
  select(BEGIN_YEARMONTH, BEGIN_DAY, BEGIN_TIME, DURATION1, STATE, INJURIES_DIRECT,
         INJURIES_INDIRECT, DEATHS_DIRECT, DEATHS_INDIRECT, DAMAGE_PROPERTY, DAMAGE_CROPS,
         TOR_F_SCALE, TOR_LENGTH, TOR_WIDTH, BEGIN_LAT, BEGIN_LON, END_LAT, END_LON, SEASON, REGION) %>%
  mutate(
    DAMAGE_PROPERTY = as.numeric(gsub("[^0-9]", "", DAMAGE_PROPERTY)), # Remove non-numeric chars
    DAMAGE_CROPS = as.numeric(gsub("[^0-9]", "", DAMAGE_CROPS)),       # Remove non-numeric chars
    TOTAL_DAMAGE = DAMAGE_PROPERTY + DAMAGE_CROPS,                      # Compute total damage
    LOG_TOTAL_DAMAGE = log1p(TOTAL_DAMAGE),                              # Log-transform to reduce skewness,
    INTENSITY = case_when(
      TOR_F_SCALE == "EF0" ~ 0,
      TOR_F_SCALE == "EF1" ~ 1,
      TOR_F_SCALE == "EF2" ~ 2,
      TOR_F_SCALE == "EF3" ~ 3,
      TOR_F_SCALE == "EF4" ~ 4,
      TOR_F_SCALE == "EF5" ~ 5,
      TRUE ~ NA_real_ # Handle unknown scales
    )
  )

# Handle missing values: Replace NAs with mean for numeric columns
tornado_data_clean <- tornado_data_clean %>%
  mutate(across(where(is.numeric), ~ ifelse(is.na(.), mean(., na.rm = TRUE), .)))

# Feature Engineering: Add binary impact indicator (e.g., high damage) and interaction terms
tornado_data_clean <- tornado_data_clean %>%
  mutate(
    HIGH_DAMAGE = ifelse(TOTAL_DAMAGE > median(TOTAL_DAMAGE, na.rm = TRUE), 1, 0),
    TOR_AREA = TOR_LENGTH * TOR_WIDTH
  )
